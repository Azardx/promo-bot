import asyncio

from scrapers.amazon import scrape as amazon
from scrapers.shopee import scrape as shopee
from scrapers.kabum import scrape as kabum

from services.filters import promo_valida


async def coletar_promos():

    scrapers = [
        amazon(),
        shopee(),
        kabum()
    ]

    results = await asyncio.gather(*scrapers, return_exceptions=True)

    promos = []
    seen = set()

    for loja in results:

        if isinstance(loja, Exception):
            continue

        for titulo, link in loja:

            if not promo_valida(titulo):
                continue

            if link in seen:
                continue

            seen.add(link)

            promos.append((titulo, link))

    return promos
